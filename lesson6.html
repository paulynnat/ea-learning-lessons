Event Listeners

Events are things that happen in the browser (clicks, key presses, mouse movements, etc.). We can "listen" for these events and respond to them.

Method 1: Inline Event Handlers

<button onclick="handleClick()">Click Me</button>

<script>
    function handleClick() {
        alert("Button clicked!");
    }
</script>

Method 2: addEventListener (Better!)

<button id="myButton">Click Me</button>

<script>
    let button = document.getElementById("myButton");
    
    button.addEventListener("click", function() {
        alert("Button clicked!");
    });
    
    // Or with a named function
    function handleClick() {
        alert("Button clicked!");
    }
    button.addEventListener("click", handleClick);
</script>

Common Events

// Click events
element.addEventListener("click", function() { });

// Mouse events
element.addEventListener("mouseenter", function() { });  // Mouse enters
element.addEventListener("mouseleave", function() { });  // Mouse leaves

// Keyboard events
element.addEventListener("keydown", function() { });     // Key pressed
element.addEventListener("keyup", function() { });       // Key released

// Form events
element.addEventListener("submit", function(e) { 
    e.preventDefault();  // Prevent form from submitting
});
element.addEventListener("input", function() { });       // Input changes

// Window events
window.addEventListener("load", function() { });         // Page loaded
window.addEventListener("resize", function() { });       // Window resized

Working with Form Inputs

Getting Input Values

// HTML
<input type="text" id="nameInput" placeholder="Enter your name">
<button id="submitBtn">Submit</button>

// JavaScript
let input = document.getElementById("nameInput");
let button = document.getElementById("submitBtn");

button.addEventListener("click", function() {
    let name = input.value;  // Get the input value
    console.log("You entered:", name);
    
    input.value = "";  // Clear the input
});

Conditional Statements

Conditionals allow your code to make decisions:

If Statements

let age = 18;

if (age >= 18) {
    console.log("You are an adult");
}

// If-else
if (age >= 18) {
    console.log("You are an adult");
} else {
    console.log("You are a minor");
}

// If-else if-else
let score = 85;

if (score >= 90) {
    console.log("Grade: A");
} else if (score >= 80) {
    console.log("Grade: B");
} else if (score >= 70) {
    console.log("Grade: C");
} else {
    console.log("Grade: F");
}

Comparison Operators

// Equality
==   // Equal to (type coercion: "5" == 5 is true)
===  // Strict equal (no type coercion: "5" === 5 is false)
!=   // Not equal
!==  // Strict not equal

// Comparison
>    // Greater than
<    // Less than
>=   // Greater than or equal
<=   // Less than or equal

// Logical operators
&&   // AND (both conditions must be true)
||   // OR (at least one condition must be true)
!    // NOT (reverses the boolean)

// Examples
if (age >= 18 && hasLicense) {
    console.log("You can drive!");
}

if (day === "Saturday" || day === "Sunday") {
    console.log("It's the weekend!");
}

Counter Demo

0

The code behind the counter:

// Counter variable
let count = 0;

// Get elements
let display = document.getElementById("counter-display");

let decrementBtn = document.getElementById("decrement-btn");
let resetBtn = document.getElementById("reset-btn");

// Increment function
incrementBtn.addEventListener("click", function() {
    count++;
    display.textContent = count;
});

// Decrement function
decrementBtn.addEventListener("click", function() {
    count--;
    display.textContent = count;
});

// Reset function
resetBtn.addEventListener("click", function() {
    count = 0;
    display.textContent = count;
});

Project 2: Todo List

Todo List Demo

    The code behind the todo list:

    // Get elements
let todoInput = document.getElementById("todo-input");
let addBtn = document.getElementById("add-todo");
let todoList = document.getElementById("todo-list");

// Add todo function
function addTodo() {
    let todoText = todoInput.value;
    
    // Check if input is empty
    if (todoText === "") {
        alert("Please enter a task!");
        return;
    }
    
    // Create list item
    let li = document.createElement("li");
    li.textContent = todoText;
    
    // Create delete button
    let deleteBtn = document.createElement("button");
    deleteBtn.textContent = "Delete";
    deleteBtn.className = "delete-btn";
    deleteBtn.addEventListener("click", function() {
        li.remove();
    });
    
    // Add delete button to list item
    li.appendChild(deleteBtn);
    
    // Add list item to list
    todoList.appendChild(li);
    
    // Clear input
    todoInput.value = "";
}

// Add click event to button
addBtn.addEventListener("click", addTodo);

// Allow Enter key to add task
todoInput.addEventListener("keypress", function(e) {
    if (e.key === "Enter") {
        addTodo();
    }
});
